Иерархия компонент

							
Ext.define('AM.view.user.Edit', {
    extend:'Ext.window.Window',
    alias:'widget.useredit',
    title:'Edit User',
    layout:'fit',
    autoShow:true,
    items: [{
                xtype:'form',
                items: [{
                        xtype:'textfield',
                        name :'name',
                        fieldLabel:'Name'
                    },
                    {
                        xtype:'textfield',
                        name :'email',
                        fieldLabel:'Email'
                    }]
    }]
});
							
						
							
Ext.define('AM.view.user.Edit', {
    extend:'Ext.window.Window',
    alias:'widget.useredit',
    title:'Edit User',
    layout:'fit',
    autoShow:true,
    initComponent: function() {
        this.items = [{
                xtype:'form',
                items: [{
                        xtype:'textfield',
                        name :'name',
                        fieldLabel:'Name'
                    },
                    {
                        xtype:'textfield',
                        name :'email',
                        fieldLabel:'Email'
                    }]
            }];
        this.callParent(arguments);
    }
});
							
						
							
Ext.define('AM.view.user.Edit', {
    extend:'Ext.window.Window',
    alias:'widget.useredit',
    title:'Edit User',
    layout:'fit',
    autoShow:true,
    initComponent: function() {
    	var me = this;
    	Ext.apply(me, {
				items: [{
	                xtype:'form',
	                items: [{
	                        xtype:'textfield',
	                        name :'name',
	                        fieldLabel:'Name'
	                    },
	                    {
	                        xtype:'textfield',
	                        name :'email',
	                        fieldLabel:'Email'
	                    }]
	            }]
				
    	});
        me.callParent(arguments);
    }
});
